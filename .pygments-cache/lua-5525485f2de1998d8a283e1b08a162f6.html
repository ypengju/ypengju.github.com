local Account = { balance = 0 }
function Account:new( o )
    o = o or {}
    setmetatable(o, self)
    self.__index = self
    return o
end</p>

<p>function Account:printBalance(  )
    print(&ldquo;Account balance = &rdquo; .. self.balance)
end</p>

<p>function Account:addBalance( v )
    if tonumber(v) then
        self.balance = self.balance + v
    end
end</p>

<p>SubAccount = Account:new()</p>

<p>function SubAccount:printSub(  )
    print(&ldquo;this is come from subAccount&rdquo;)
end</p>

<p>a = SubAccount:new()
a:printSub()        &ndash; this is come from subAccount
a:printBalance()    &ndash; Account balance = 0
a:addBalance(10)
a:printBalance()    &ndash; Account balance = 10 <br/>
