local Account = {}
function Account:new( o )
	o = o or {}
	setmetatable(o, self)
	self.__index = self
	return o
end
function Account:printBalance(  )
	print("Account balance = " .. self.balance)
end
a = Account:new{ balance = 1 }
b = Account:new{ balance = 2}
a:printBalance() -- Account balance = 1
b:printBalance() -- Account balance = 2
