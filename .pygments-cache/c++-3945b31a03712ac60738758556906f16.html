#include <iostream>
#include "lua.hpp"

int main(int argc, const char * argv[]) {
    
    //1. 初始化Lua虚拟机，初始话新lua栈
    lua_State *lua_state;
    lua_state = luaL_newstate();
    
    //2. 打开所有的标准库
    luaL_openlibs(lua_state);
    
    //3、运行hello.lua脚本
    luaL_dofile(lua_state, "hello.lua");
    
    //4. 关闭Lua虚拟机
    lua_close(lua_state);
    
}
