mt.__le = function ( a, b )
	for k,v in pairs(a) do
		if  not b[k] then return false end
	end
	return true
end
mt.__lt = function ( a, b )
	return a <= b and not (b <= a)
end
t1 = new{10, 20, 30}
t2 = new{10, 20}
print(t1 < t2)  --false
print(t2 < t1)	--true
print(t1 > t2)  --true
print(t1 >= t2) --true
